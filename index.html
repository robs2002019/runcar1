═══════════════════════════════════════════════════════════════ # CONFIG # ═══════════════════════════════════════════════════════════════ SECRET_KEY = os.getenv("SECRET_KEY") if not SECRET_KEY: raise RuntimeError("SECRET_KEY no configurada") ALGORITHM = "HS256" ACCESS_TOKEN_EXPIRE_DAYS = 7 APP_COMMISSION_RATE = 0.08 VIAJE_EXPIRY_SECONDS = 200 MAX_IMAGE_SIZE = 5 * 1024 * 1024 FIREBASE_CRED_PATH = os.getenv( "FIREBASE_CREDENTIALS_JSON", "serviceAccountKey.json" ) # ═══════════════════════════════════════════════════════════════ # RATE LIMITER # ═══════════════════════════════════════════════════════════════ limiter = Limiter(key_func=get_remote_address) # ═══════════════════════════════════════════════════════════════ # FIREBASE # ═══════════════════════════════════════════════════════════════ def init_firebase(): if not firebase_admin._apps: if not os.path.exists(FIREBASE_CRED_PATH): raise RuntimeError( "No existe FIREBASE_CREDENTIALS_JSON" ) cred = credentials.Certificate(FIREBASE_CRED_PATH) firebase_admin.initialize_app(cred) return firestore.client() db = init_firebase() # ═══════════════════════════════════════════════════════════════ # APP # ═══════════════════════════════════════════════════════════════ app = FastAPI( title="RUN CAR PRO MAX API", version="2.0.0" ) app.state.limiter = limiter app.add_exception_handler( RateLimitExceeded, lambda request, exc: HTTPException( status_code=429, detail="Demasiadas solicitudes" ) ) app.add_middleware(SlowAPIMiddleware) # ═══════════════════════════════════════════════════════════════ # CORS # ═══════════════════════════════════════════════════════════════ ALLOWED_ORIGINS = [ "http://localhost:3000", "https://runcarpro.com" "https://robs2002019.github.io/runcar1/" ] app.add_middleware( CORSMiddleware, allow_origins=ALLOWED_ORIGINS, allow_credentials=True, allow_methods=["*"], allow_headers=["*"] ) # ═══════════════════════════════════════════════════════════════ # AUTH # ═══════════════════════════════════════════════════════════════ pwd_context = CryptContext( schemes=["bcrypt"], deprecated="auto" ) oauth2_scheme = OAuth2PasswordBearer( tokenUrl="/auth/login" ) # ═══════════════════════════════════════════════════════════════ # HELPERS # ═══════════════════════════════════════════════════════════════ def hash_password(password: str): return pwd_context.hash(password) def verify_password(password, hashed): return pwd_context.verify(password, hashed) def create_access_token(data: dict): to_encode = data.copy() expire = datetime.utcnow() + timedelta( days=ACCESS_TOKEN_EXPIRE_DAYS ) to_encode.update({ "exp": expire, "iat": datetime.utcnow() }) return jwt.encode( to_encode, SECRET_KEY, algorithm=ALGORITHM ) def validate_password(password: str): if len(password) < 8: raise HTTPException( 400, "Contraseña mínima: 8 caracteres" ) if not re.search(r"[A-Z]", password): raise HTTPException( 400, "Debe contener una mayúscula" ) if not re.search(r"\d", password): raise HTTPException( 400, "Debe contener un número" ) def validate_base64_image(data: str): if not data: return try: decoded = base64.b64decode(data) if len(decoded) > MAX_IMAGE_SIZE: raise HTTPException( 400, "Imagen demasiado grande" ) except Exception: raise HTTPException( 400, "Imagen inválida" ) # ═══════════════════════════════════════════════════════════════ # USER AUTH # ═══════════════════════════════════════════════════════════════ async def get_current_user( token: str = Depends(oauth2_scheme) ): credentials_exception = HTTPException( status_code=401, detail="Token inválido" ) try: payload = jwt.decode( token, SECRET_KEY, algorithms=[ALGORITHM] ) uid = payload.get("sub") if uid is None: raise credentials_exception except JWTError: raise credentials_exception snap = db.collection( "usuarios_privados" ).document(uid).get() if not snap.exists: raise credentials_exception user = snap.to_dict() if user.get("estado") != "activo": raise HTTPException( 403, "Cuenta inactiva" ) return user def require_role(*roles): async def checker( user: dict = Depends(get_current_user) ): if user.get("rol") not in roles: raise HTTPException( 403, "Acceso denegado" ) return user return checker # ═══════════════════════════════════════════════════════════════ # MODELS # ═══════════════════════════════════════════════════════════════ class RegisterRequest(BaseModel): email: EmailStr password: str nombre: str = Field( min_length=2, max_length=100 ) telefono: Optional[str] = "" rol: str = "cliente" placa: Optional[str] = "" marca: Optional[str] = "" modelo: Optional[str] = "" anio: Optional[str] = "" foto_perfil: Optional[str] = "" licencia: Optional[str] = "" penales: Optional[str] = "" solvencia: Optional[str] = "" foto_placa: Optional[str] = "" foto_carro1: Optional[str] = "" foto_carro2: Optional[str] = "" foto_carro3: Optional[str] = "" class LoginResponse(BaseModel): access_token: str token_type: str uid: str rol: str class PublicarViajeRequest(BaseModel): origen: str destino: str oferta_original: float personas: int = 1 bebes: int = 0 mascotas: int = 0 notas: Optional[str] = "" # ═══════════════════════════════════════════════════════════════ # AUTH ENDPOINTS # ═══════════════════════════════════════════════════════════════ @app.post( "/auth/register", tags=["Auth"] ) async def register(body: RegisterRequest): validate_password(body.password) validate_base64_image(body.foto_perfil) validate_base64_image(body.licencia) validate_base64_image(body.penales) validate_base64_image(body.solvencia) existing = ( db.collection("usuarios_privados") .where("email", "==", body.email) .limit(1) .get() ) if existing: raise HTTPException( 400, "Correo ya registrado" ) uid = hashlib.sha256( f"{body.email}{datetime.utcnow()}".encode() ).hexdigest()[:28] password_hash = hash_password(body.password) public_user = { "uid": uid, "nombre": body.nombre, "telefono": body.telefono, "rol": body.rol, "estado": "activo", "foto_perfil": body.foto_perfil, "timestamp": datetime.utcnow().isoformat() } private_user = { **public_user, "email": body.email, "password_hash": password_hash } if body.rol == "driver": public_user.update({ "placa": body.placa, "marca": body.marca, "modelo": body.modelo, "anio": body.anio }) private_user.update({ "placa": body.placa, "marca": body.marca, "modelo": body.modelo, "anio": body.anio, "licencia": body.licencia, "penales": body.penales, "solvencia": body.solvencia, "foto_placa": body.foto_placa, "foto_carro1": body.foto_carro1, "foto_carro2": body.foto_carro2, "foto_carro3": body.foto_carro3 }) db.collection( "usuarios_publicos" ).document(uid).set(public_user) db.collection( "usuarios_privados" ).document(uid).set(private_user) token = create_access_token({ "sub": uid, "rol": body.rol }) logger.info(f"Usuario registrado: {uid}") return { "access_token": token, "token_type": "bearer", "uid": uid, "rol": body.rol } # ═══════════════════════════════════════════════════════════════ # LOGIN # ═══════════════════════════════════════════════════════════════ @app.post( "/auth/login", response_model=LoginResponse, tags=["Auth"] ) @limiter.limit("5/minute") async def login( request: Request, form_data: OAuth2PasswordRequestForm = Depends() ): docs = ( db.collection("usuarios_privados") .where("email", "==", form_data.username) .limit(1) .get() ) if not docs: raise HTTPException( 401, "Credenciales inválidas" ) user = docs[0].to_dict() if not verify_password( form_data.password, user["password_hash"] ): raise HTTPException( 401, "Credenciales inválidas" ) token = create_access_token({ "sub": user["uid"], "rol": user["rol"] }) logger.info(f"Login exitoso: {user['uid']}") return { "access_token": token, "token_type": "bearer", "uid": user["uid"], "rol": user["rol"] } # ═══════════════════════════════════════════════════════════════ # PERFIL # ═══════════════════════════════════════════════════════════════ @app.get( "/auth/me", tags=["Auth"] ) async def me( user: dict = Depends(get_current_user) ): user.pop("password_hash", None) return user # ═══════════════════════════════════════════════════════════════ # VIAJES # ═══════════════════════════════════════════════════════════════ @app.post( "/viajes", tags=["Viajes"] ) async def publicar_viaje( body: PublicarViajeRequest, user: dict = Depends(require_role("cliente")) ): viaje = { "clienteId": user["uid"], "clienteNombre": user["nombre"], "origen": body.origen, "destino": body.destino, "personas": body.personas, "bebes": body.bebes, "mascotas": body.mascotas, "notas": body.notas, "ofertaOriginal": body.oferta_original, "ofertaActual": body.oferta_original, "estado": "buscando", "timestamp": datetime.utcnow().isoformat() } ref = db.collection( "viajes" ).add(viaje) viaje_id = ref[1].id logger.info(f"Nuevo viaje: {viaje_id}") return { "id": viaje_id, **viaje } # ═══════════════════════════════════════════════════════════════ # LISTAR VIAJES # ═══════════════════════════════════════════════════════════════ @app.get( "/viajes", tags=["Viajes"] ) async def listar_viajes( user: dict = Depends( require_role("driver", "admin") ) ): docs = ( db.collection("viajes") .limit(20) .get() ) viajes = [] now = datetime.utcnow() for d in docs: v = d.to_dict() ts = datetime.fromisoformat( v["timestamp"] ) elapsed = ( now - ts ).total_seconds() if elapsed > VIAJE_EXPIRY_SECONDS: db.collection( "viajes" ).document(d.id).update({ "estado": "expirado" }) continue if v["estado"] in [ "buscando", "contraoferta" ]: v["id"] = d.id viajes.append(v) return viajes # ═══════════════════════════════════════════════════════════════ # CHAT # ═══════════════════════════════════════════════════════════════ class ChatRequest(BaseModel): msg: str LAST_MESSAGE_TIME = {} @app.post( "/viajes/{viaje_id}/mensajes", tags=["Chat"] ) async def enviar_mensaje( viaje_id: str, body: ChatRequest, user: dict = Depends(get_current_user) ): now = datetime.utcnow() last = LAST_MESSAGE_TIME.get( user["uid"] ) if last: diff = ( now - last ).total_seconds() if diff < 1: raise HTTPException( 429, "Muy rápido" ) LAST_MESSAGE_TIME[ user["uid"] ] = now msg = { "uid": user["uid"], "nombre": user["nombre"], "msg": body.msg.strip(), "timestamp": now.isoformat() } db.collection( "viajes" ).document(viaje_id).collection( "mensajes" ).add(msg) return {"ok": True} # ═══════════════════════════════════════════════════════════════ # GANANCIAS # ═══════════════════════════════════════════════════════════════ @app.get( "/ganancias/me", tags=["Ganancias"] ) async def ganancias( user: dict = Depends( require_role("driver") ) ): docs = ( db.collection("ganancias") .where( "driverId", "==", user["uid"] ) .limit(50) .get() ) registros = [] total = 0 for d in docs: r = d.to_dict() registros.append(r) total += float( r.get("monto", 0) ) comision = round( total * APP_COMMISSION_RATE, 2 ) neto = round( total - comision, 2 ) return { "registros": registros, "bruto": round(total, 2), "comision": comision, "neto": neto } #
